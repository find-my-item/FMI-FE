import { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: "*",
      allow: "/",
      disallow: ["/admin", "/mypage/notice", "/mypage/inquiries", "/chat/"],
    },
    sitemap: "https://www.finditem.kr/sitemap.xml",
  };
}
